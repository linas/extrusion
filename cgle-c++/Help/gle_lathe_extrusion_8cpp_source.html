64 
65  double localup[3];
66  double len;
67  double trans[2];
68  double delt[2][3];
69 
70  if (m_ptrUp[1] != 0.0)
71  {
72  localup[0] = m_ptrUp[0];
73  localup[1] = 0.0;
74  localup[2] = m_ptrUp[2];
75  VEC_LENGTH(len, localup);
76  if (len != 0.0)
77  {
78  len = 1.0 / len;
79  localup[0] *= len;
80  localup[2] *= len;
81  VEC_SCALE(localup, len, localup);
82  }
83  else
84  {
85  // invalid up vector was passed in
86  localup[0] = 0.0;
87  localup[2] = 1.0;
88  }
89  }
90  else
91  {
92  VEC_COPY(localup, m_ptrUp);
93  }
94 
95  for (int i = 0; i < 2; i++)
96  {
97  for (int j = 0; j < 3; j++)
98  {
99  delt[i][j] = FXForm[i][j];
100  }
101  }
102 
103  trans[0] = localup[2] * dRadiusdTheta - localup[0] * dZdTheta;
104  trans[1] = localup[0] * dRadiusdTheta + localup[2] * dZdTheta;
105 
106  // now, add this translation vector into the affine xform
107  delt[0][2] += trans[0];
108  delt[1][2] += trans[1];
109 
110  CgleSpiralExtrusion::SetTForm(delt);
111 
112  CgleSpiralExtrusion::Draw(StartRadius, 0.0, StartZ, 0.0, StartTheta,
