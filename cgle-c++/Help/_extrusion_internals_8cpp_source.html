117 //----------------------------------------------------------------------------
118 inline void sphere_texgen (double x, double y, double z,
119  int jcnt, int which_end)
120 {
121  double theta, phi;
122 
123  // let phi and theta range fro 0 to 1
124  phi = 0.5 * atan2 (x, y) / M_PI;
125  phi += 0.5;
126 
127  theta = 1.0 - acos (z) / M_PI;
128 
129  // if first vertex, merely record the texture coords
130  if (num_vert == 0)
131  {
132  prev_x = phi;
133  prev_y = theta;
134  num_vert ++;
135  }
136  else
137  {
138 
139  // if texture coordinates changed radically, wrap them
140  if ((prev_y - theta) > 0.6)
141  {
142  theta +=1.0;
143  }
144  else if ((prev_y - theta) < -0.6)
145  {
146  theta -=1.0;
147  } // else no-op
148  prev_y = theta;
149 
150 
151  // if texture coordinates changed radically, wrap them
152  if ((prev_x - phi) > 0.6)
153  {
154  phi +=1.0;
155  }
156  else if ((prev_x - phi) < -0.6)
157  {
158  phi -=1.0;
159  } // else no-op
160  prev_x = phi;
161 
162  }
163 
164  T2F_D (phi, theta);
165 }

416 //----------------------------------------------------------------------------
417 //
418  // The uviewdirection subroutine computes and returns a 4x4 rotation
419  // matrix that puts the negative z axis along the direction v21 and
420  // puts the y axis along the up vector.
421  //
422  // Note that this code is fairly tolerant of "weird" paramters.
423  // It normalizes when necessary, it does nothing when vectors are of
424  // zero length, or are co-linear. This code shouldn't croak, no matter
425  // what the user sends in as arguments.
426  //
427 void uview_direction_d (double m[4][4], // returned
428  double v21[3], //input
429  double up[3]) // input
430 {
431  double amat[4][4];
432  double bmat[4][4];
433  double cmat[4][4];
434  double v_hat_21[3];
435  double v_xy[3];
436  double sine, cosine;
437  double len;
438  double up_proj[3];
439  double tmp[3];
440 
441  // find the unit vector that points in the v21 direction
442  VEC_COPY (v_hat_21, v21);
443  VEC_LENGTH (len, v_hat_21);
444  if (len != 0.0) {
445  len = 1.0 / len;
446  VEC_SCALE (v_hat_21, len, v_hat_21);
447 
448  // rotate z in the xz-plane until same latitude
449  sine = sqrt ( 1.0 - v_hat_21[2] * v_hat_21[2]);
450  ROTY_CS (amat, (-v_hat_21[2]), (-sine));
451 
452  } else {
453 
454  // error condition: zero length vecotr passed in -- do nothing
455  IDENTIFY_MATRIX_4X4 (amat);
456  }
457 
458 
459  // project v21 onto the xy plane
460  v_xy[0] = v21[0];
461  v_xy[1] = v21[1];
462  v_xy[2] = 0.0;
463  VEC_LENGTH (len, v_xy);
464 
465  // rotate in the x-y plane until v21 lies on z axis ---
466  // but of course, if its already there, do nothing
467  if (len != 0.0) {
468 
469  // want xy projection to be unit vector, so that sines/cosines pop out
470  len = 1.0 / len;
471  VEC_SCALE (v_xy, len, v_xy);
472 
473  // rotate the projection of v21 in the xy-plane over to the x axis
474  ROTZ_CS (bmat, v_xy[0], v_xy[1]);
475 
476  // concatenate these together
477  MATRIX_PRODUCT_4X4 (cmat, amat, bmat);
478 
479  } else {
480 
481  // no-op -- vector is already in correct position
482  COPY_MATRIX_4X4 (cmat, amat);
483  }
484 
485  // up vector really should be perpendicular to the x-form direction --
486  // Use up a couple of cycles, and make sure it is,
487  // just in case the user blew it.
488  //
489  VEC_PERP (up_proj, up, v_hat_21);
490  VEC_LENGTH (len, up_proj);
491  if (len != 0.0) {
492 
493  // normalize the vector
494  len = 1.0/len;
495  VEC_SCALE (up_proj, len, up_proj);
496 
497  // compare the up-vector to the y-axis to get the cosine of the angle
498  tmp [0] = cmat [1][0];
499  tmp [1] = cmat [1][1];
500  tmp [2] = cmat [1][2];
501  VEC_DOT_PRODUCT (cosine, tmp, up_proj);
502 
503  // compare the up-vector to the x-axis to get the sine of the angle
504  tmp [0] = cmat [0][0];
505  tmp [1] = cmat [0][1];
506  tmp [2] = cmat [0][2];
507  VEC_DOT_PRODUCT (sine, tmp, up_proj);
508 
509  // rotate to align the up vector with the y-axis
510  ROTZ_CS (amat, cosine, -sine);
511 
512  // This xform, although computed last, acts first
513  MATRIX_PRODUCT_4X4 (m, amat, cmat);
514 
515  } else {
516 
517  // error condition: up vector is indeterminate (zero length)
518  // -- do nothing
519  COPY_MATRIX_4X4 (m, cmat);
520  }
521 }

526  // The uviewpoint subroutine computes and returns a 4x4 matrix that
527  // translates the origen to the point v1, puts the negative z axis
528  // along the direction v21==v2-v1, and puts the y axis along the up
529  // vector.

531 void uviewpoint_d (double m[4][4], // returned
532  double v1[3], // input
533  double v2[3], // input
534  double up[3]) // input
535 {
536  double v_hat_21 [3];

538  double rot_mat[4][4];

540  VEC_DIFF (v_hat_21, v2, v1);
541 
542  // compute rotation matrix that takes -z axis to the v21 axis,
543  // and y to the up dierction
544  uview_direction_d (rot_mat, v_hat_21, up);
545

553  MATRIX_PRODUCT_4X4 (m, rot_mat, trans_mat);

559 void urot_axis_d (double m[4][4], // returned
560  double omega, // input
561  double axis[3]) // input
562 {
563  double s, c, ssq, csq, cts;
564  double tmp;
565 
566  //
567  // The formula coded up below can be derived by using the
568  // homomorphism between SU(2) and O(3), namely, that the
569  // 3x3 rotation matrix R is given by
570  // t.R.v = S(-1) t.v S
571  // where
572  // t are the Pauli matrices (similar to Quaternions, easier to use)
573  // v is an arbitrary 3-vector
574  // and S is a 2x2 hermitian matrix:
575  // S = exp ( i omega t.axis / 2 )
576  //
577  // (Also, remember that computer graphics uses the transpose of R).
578  //
579  // The Pauli matrices are:
580  //
581  // tx = (0 1) ty = (0 -i) tz = (1 0)
582  // (1 0) (i 0) (0 -1)
583  //
584  // Note that no error checking is done -- if the axis vector is not
585  // of unit length, you'll get strange results.
586  //
587 
588  tmp = (double) omega / 2.0;
589  s = sin (tmp);
590  c = cos (tmp);
591 
592  ssq = s*s;
593  csq = c*c;
594 
595  m[0][0] = m[1][1] = m[2][2] = csq - ssq;
596 
597  ssq *= 2.0;
598 
599  // on-diagonal entries
600  m[0][0] += ssq * axis[0]*axis[0];
601  m[1][1] += ssq * axis[1]*axis[1];
602  m[2][2] += ssq * axis[2]*axis[2];
603 
604  // off-diagonal entries
605  m[0][1] = m[1][0] = axis[0] * axis[1] * ssq;
606  m[1][2] = m[2][1] = axis[1] * axis[2] * ssq;
607  m[2][0] = m[0][2] = axis[2] * axis[0] * ssq;
608 
609  cts = 2.0 * c * s;
610 
611  tmp = cts * axis[2];
612  m[0][1] += tmp;
613  m[1][0] -= tmp;
614 
615  tmp = cts * axis[0];
616  m[1][2] += tmp;
617  m[2][1] -= tmp;
618 
619  tmp = cts * axis[1];
620  m[2][0] += tmp;
621  m[0][2] -= tmp;
622 
623  // homogeneous entries
624  m[0][3] = m[1][3] = m[2][3] = m[3][2] = m[3][1] = m[3][0] = 0.0;
625  m[3][3] = 1.0;
626 
627 
628 }

630 void urot_about_axis_d (double m[4][4], // returned
631  double angle, // input
632  double axis[3]) // input
633 {
634  gutDouble len, ax[3];
635 
636  angle *= M_PI/180.0; // convert to radians
637 
638  // renormalize axis vector, if needed
639  len = axis[0]*axis[0] + axis[1]*axis[1] + axis[2]*axis[2];
640 
641  // we can save some machine instructions by normalizing only
642  // if needed. The compiler should be able to schedule in the
643  // if test "for free".
644  if (len != 1.0) {
645  len = (gutDouble) (1.0 / sqrt ((double) len));
646  ax[0] = axis[0] * len;
647  ax[1] = axis[1] * len;
648  ax[2] = axis[2] * len;
649  urot_axis_d (m, angle, (double*)ax);
650  } else {
651  urot_axis_d (m, angle, (double*)axis);
652  }
653 }
