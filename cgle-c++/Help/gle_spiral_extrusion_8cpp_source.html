23 // The spiral primitive forms the basis for the helicoid primitive.
24 //
25 // Note that this primitive sweeps a contour along a helical path.
26 // The algorithm assumes that the path is embedded in Euclidean space,
27 // and uses parallel transport along the path. Parallel transport
28 // provides the simplest mathematical model for moving a coordinate
29 // system along a curved path, but some of the effects of doing so
30 // may prove to be surprising to one uninitiated to the concept.
31 //
32 // Thus, we provide another, related, algorithm below, called "lathe"
33 // which introduces a torsion component along the path, correcting for
34 // the rotation induced by the helical path.
35 //
36 // If the above sounds like gobldy-gook to you, you may want to brush
37 // up on differential geometry. Recommend Spivak, Differential Geometry,
38 // Volume 1, pages xx-xx.
39 //----------------------------------------------------------------------------

95 void CgleSpiralExtrusion::SetTForm(double dXFormdTheta[2][3])
96 {
97  double mA[2][2], mB[2][2];
98  double run[2][2];
99  double deltaTrans[2];
100  double trans[2];
101  double delta = FDeltaAngle / (2.0 * M_PI);
102 
103 
104  // If there is a deformation matrix specified, then a deformation
105  // path must be generated also
106  if (FStartXForm[0][0] != -1E300)
107  {
108  if (dXFormdTheta == NULL)
109  {
110  for (int i = 0; i<m_iPoints; i++)
111  {
112  m_ptrXforms[i][0][0] = FStartXForm[0][0];
113  m_ptrXforms[i][0][1] = FStartXForm[0][1];
114  m_ptrXforms[i][0][2] = FStartXForm[0][2];
115  m_ptrXforms[i][1][0] = FStartXForm[1][0];
116  m_ptrXforms[i][1][1] = FStartXForm[1][1];
117  m_ptrXforms[i][1][2] = FStartXForm[1][2];
118  }
119  }
120  else
121  {
122  //
123  // if there is a differential matrix specified, treat it a
124  // a tangent (algebraic, infinitessimal) matrix. We need to
125  // project it into the group of real 2x2 matricies. (Note that
126  // the specified matrix is affine. We treat the translation
127  // components linearly, and only treat the 2x2 submatrix as an
128  // algebraic tangenet).
129  //
130  // For exponentiaition, we use the well known approx:
131  // exp(x) = lim (N->inf) (1+x/N) ** N
132  // and take N=32.
133  //
134 
135  // initialize translation and delta translation
136  deltaTrans[0] = delta * dXFormdTheta[0][2];
137  deltaTrans[1] = delta * dXFormdTheta[1][2];
138  trans[0] = FStartXForm[0][2];
139  trans[1] = FStartXForm[1][2];
140 
141  // prepare the tangent matrix
142  delta /= 32.0;
143  mA[0][0] = 1.0 + delta * dXFormdTheta[0][0];
144  mA[0][1] = delta * dXFormdTheta[0][1];
145  mA[1][0] = delta * dXFormdTheta[1][0];
146  mA[1][1] = 1.0 + delta * dXFormdTheta[1][1];
147 
148  // compute exponential of matrix
149  MATRIX_PRODUCT_2X2(mB, mA, mA); // squared
150  MATRIX_PRODUCT_2X2(mA, mB, mB); // 4th power
151  MATRIX_PRODUCT_2X2(mB, mA, mA); // 8th power
152  MATRIX_PRODUCT_2X2(mA, mB, mB); // 16th power
153  MATRIX_PRODUCT_2X2(mB, mA, mA); // 32nd power
154 
155  // initialize running matrix
156  COPY_MATRIX_2X2(run, FStartXForm);
157 
158  // remember, the first point is hidden -- load some, any
159  // xform for the first point
160  m_ptrXforms[0][0][0] = FStartXForm[0][0];
161  m_ptrXforms[0][0][1] = FStartXForm[0][1];
162  m_ptrXforms[0][0][2] = FStartXForm[0][2];
163  m_ptrXforms[0][1][0] = FStartXForm[1][0];
164  m_ptrXforms[0][1][1] = FStartXForm[1][1];
165  m_ptrXforms[0][1][2] = FStartXForm[1][2];
166 
167  for (int i = 1; i<m_iPoints; i++)
168  {
169  m_ptrXforms[i][0][0] = run[0][0];
170  m_ptrXforms[i][0][1] = run[0][1];
171  m_ptrXforms[i][1][0] = run[1][0];
172  m_ptrXforms[i][1][1] = run[1][1];
173 
174  // integrate to get exponential matrix
175  // (Note that the group action is a left-action --
176  // i.e. multiply on the left (not the right))
177  MATRIX_PRODUCT_2X2(mA, mB, run);
178  COPY_MATRIX_2X2(run, mA);
179 
180  m_ptrXforms[i][0][2] = trans[0];
181  m_ptrXforms[i][1][2] = trans[1];
182  trans[0] += deltaTrans[0];
183  trans[1] += deltaTrans[1];
184 
185  }
186  }
187  }
188 }

190 void CgleSpiralExtrusion::Draw(double StartRadius,
191  double dRadiusdTheta, double StartZ,
192  double dZdTheta, double StartTheta,
193  float Color_Array[][3], bool bTextured)
194 {
195 
196  int saved_style;
197  double delta, cprev, sprev, cdelta, sdelta, ccurr, scurr;
198 
199  StartTheta *= M_PI / 180.0;
200  StartTheta -= FDeltaAngle;
201 
202  // initialize factors
203  cprev = cos((double)StartTheta);
204  sprev = sin((double)StartTheta);
205 
206  cdelta = cos((double)FDeltaAngle);
207  sdelta = sin((double)FDeltaAngle);
208 
209  // renormalize differential factors
210  delta = FDeltaAngle / (2.0 * M_PI);
211  dZdTheta *= delta;
212  dRadiusdTheta *= delta;
213 
214  // remember, the first point is hidden, so back-step
215  StartZ -= dZdTheta;
216  StartRadius -= dRadiusdTheta;
217 
218  // draw spiral path using recursion relations for sine, cosine *
219  for (int i = 0; i<m_iPoints; i++)
220  {
221  FPts[3 * i] = StartRadius * cprev;
222  FPts[3 * i + 1] = StartRadius * sprev;
223  FPts[3 * i + 2] = (double)StartZ;
224 
225  StartZ += dZdTheta;
226  StartRadius += dRadiusdTheta;
227  ccurr = cprev * cdelta - sprev * sdelta;
228  scurr = cprev * sdelta + sprev * cdelta;
229  cprev = ccurr;
230  sprev = scurr;
231  }
232 
233  // save the current join style
234  saved_style = CgleBaseExtrusion::GetExtrusionMode();//extrusion_join_style;
235 
236  // Allow only angle joins (for performance reasons).
237  // The idea is that if the tesselation is fine enough, then an angle
238  // join should be sufficient to get the desired visual quality. A
239  // raw join would look terrible, an cut join would leave garbage
240  // everywhere, and a round join will over-tesselate (and thus
241  // should be avoided for performance reasons).
242  //
243 
244  int style = saved_style & ~GLE_JN_MASK;
245  style |= GLE_JN_ANGLE;
246 
247  CgleBaseExtrusion::SetExtrusionMode(style);
248 
249  CgleBaseExtrusion::DrawExtrusion((double(*)[3])FPts, Color_Array, bTextured);
250 
251 }

gleTexMode::GLE_TEXTURE_NORMAL_MODEL_CYL
Similar to GLE_TEXTURE_NORMAL_CYL, except that the untransformed normals are used. As a result, textures tends to stick to the extrusion according to the extrusions local surface coordinates rather than according to real-space coordinates. This will in general provide the correct style of texture mapping when affine transforms are being applied to the contour, since the coordinates used are those prior to the affine transform.
