38 void CgleHelicoidExtrusion::Draw(double CrossSectionRadius, double StartRadius,
39  double dRadiusdTheta, double StartZ,
40  double dZdTheta, double StartTheta,
41  float Color_Array[][3], bool bTextured)
42 {
43 
44  double c, s;
45 
46 
47  // initialize sine and cosine for circle recursion equations
48  s = sin(2.0*M_PI / ((double)FFacets));
49  c = cos(2.0*M_PI / ((double)FFacets));
50 
51  m_ptrCont_Norms[0][0] = 1.0;
52  m_ptrCont_Norms[0][1] = 0.0;
53  m_ptrContours[0][0] = CrossSectionRadius;
54  m_ptrContours[0][1] = 0.0;
55 
56  // draw a norm using recursion relations
57  for (int i = 1; i< FFacets + 1; i++)
58  {
59  m_ptrCont_Norms[i][0] = m_ptrCont_Norms[i - 1][0] * c - m_ptrCont_Norms[i - 1][1] * s;
60  m_ptrCont_Norms[i][1] = m_ptrCont_Norms[i - 1][0] * s + m_ptrCont_Norms[i - 1][1] * c;
61  m_ptrContours[i][0] = CrossSectionRadius * m_ptrCont_Norms[i][0];
62  m_ptrContours[i][1] = CrossSectionRadius * m_ptrCont_Norms[i][1];
63  }
